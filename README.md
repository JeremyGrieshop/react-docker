# docker

## Install

```bash
npm install --save docker
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'docker'
import 'docker/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
